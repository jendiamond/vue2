1. Start the Effect with the Button.
The Effect should alternate the 'highlight' or 'shrink' class on each new setInterval tick
(attach respective class to the div with id 'effect' below)
Refresh the page to stop the effect.

index.html


            <button @click='startEffect'>Start Effect</button>
            <div id='effect' v-bind:class='selectClass'></div>

main.js


            new Vue({
              el: '#app',
              data: {
                selectClass: {
                  shrink: true,
                  highlight: false,
                },
              },

              methods: {
                startEffect: function () {
                  var vm = this;
                  setInterval(function () {
                    vm.selectClass.shrink = !vm.selectClass.shrink;
                    vm.selectClass.highlight = !vm.selectClass.highlight;
                  }, 3000);
                },
              }

styles.css


            #effect {
              width: 400px;
              height: 200px;
              border: 1px solid black;
              border-radius: 50%;
            }

            .highlight {
              background-color: #FF69B4;
              width: 500px !important;
            }

            .shrink {
              background-color: pink;
              width: 200px !important;
            }
