index.html

<div class='demo' @click="attachOrange = !attachOrange" :class="{orange: attachOrange}"></div>

styles.css


          .demo {
            width: 60px;
            height: 60px;
            border: 1px solid gray;
            display: inline-block;
            margin: 10px;
            border-radius: 50%;
          }
          .red {
            background-color: red;
          }
          .orange {
            background-color: #FF7F00;
          }
          .yellow {
            background-color: yellow;
          }
          .green {
            background-color: #37b206;
          }
          .blue {
            background-color: #2165d8;
          }
          .indigo {
            background-color: indigo;
          }
          .violet {
            background-color: #8A2BE2;
          }

main.js


        new Vue({
          el: '#app',
          data: {
            attachRed: false,
            attachOrange: false,
            attachYellow: false,
            attachGreen: false,
            attachBlue: false,
            attachIndigo: false,
            attachViolet: false,
          }
        });
