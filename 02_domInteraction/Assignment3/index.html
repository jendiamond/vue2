Print 'done' once you did it

Current Value: {{ value }}

<p>Result: { { result() } }</p> <p>Current Value: { { value } }</p> <button @click="value += 5">Add 5</button> <button @click="value += 1">Add 1</button> <p>Result: { { result() } }</p>

2. Watch for changes in the "result"

and reset the "value" after 5 seconds

(hint: setTimeout(..., 5000)

Reset Value: {{ value }}

new Vue({
  el: '#exercise',
  data: {
    value: 0,
  },
  computed: {
    output: function(value) {
      return this.value < 37 ? 'not there yet' : 'done';
    }
  },
  watch: {
    value: function(value) {
      var vm = this;
      setTimeout(function(){
        vm.value = 0;
      }
    }
  },
  methods: {
    result: function(value) {
      return this.value < 37 ? 'not there yet' : 'done';
    }
  }
});
